                                     fullname      text NOT NULL,
                                     "position"    varchar(100),

);

-- Процедура UpdateTaskStatus
CREATE OR REPLACE PROCEDURE tasks."UpdateTaskStatus"(
    IN p_task_id integer,
    IN p_new_status integer,
    IN p_changed_by_id integer,
    IN p_reason text DEFAULT NULL
)
    LANGUAGE plpgsql
AS $$
DECLARE
    v_old_status_id INT;
BEGIN
    IF NOT EXISTS (SELECT * FROM tasks."Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;

    SELECT status_id INTO v_old_status_id FROM tasks."Tasks" WHERE id = p_task_id;
    IF v_old_status_id = p_new_status THEN
        RAISE NOTICE 'Статус задачи % не изменился', p_task_id;
        RETURN;
    END IF;
    UPDATE tasks."Tasks"
    SET status_id = p_new_status,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;
    INSERT INTO audit."TaskStatusHistory"(
        task_id, old_status_id, new_status_id, changed_by_id, reason, changed_at
    ) VALUES (
                 p_task_id, v_old_status_id, p_new_status, p_changed_by_id, p_reason, CURRENT_TIMESTAMP
             );
END;
$$;
ALTER PROCEDURE tasks."UpdateTaskStatus"(integer, integer, integer, text) OWNER TO postgres;

-- Процедура UpdateTaskDeadline
CREATE OR REPLACE PROCEDURE tasks."UpdateTaskDeadline"(
    IN p_task_id integer,
    IN p_new_deadline date,
    IN p_changed_by_id integer,
    IN p_reason text DEFAULT NULL
)
    LANGUAGE plpgsql
AS $$
DECLARE
    v_old_deadline DATE;
BEGIN
    IF NOT EXISTS (SELECT 1 FROM tasks."Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;
    SELECT due_date INTO v_old_deadline FROM tasks."Tasks" WHERE id = p_task_id;
    IF v_old_deadline = p_new_deadline THEN
        RAISE NOTICE 'Дедлайн задачи % не изменился', p_task_id;
        RETURN;
    END IF;
    UPDATE tasks."Tasks"
    SET due_date = p_new_deadline,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;
    INSERT INTO audit."TaskDeadLineHistory"(
        task_id, old_due_date, new_due_date, changed_by_id, reason, changed_at
    ) VALUES (
                 p_task_id, v_old_deadline, p_new_deadline, p_changed_by_id, p_reason, CURRENT_TIMESTAMP
             );
END;
$$;
ALTER PROCEDURE tasks."UpdateTaskDeadline"(integer, date, integer, text) OWNER TO postgres;

-- Процедура UpdateTask
CREATE OR REPLACE PROCEDURE tasks."UpdateTask"(
    IN p_id integer,
    IN p_title varchar DEFAULT NULL,
    IN p_description text DEFAULT NULL,
    IN p_status_id integer DEFAULT NULL,
    IN p_create_date date DEFAULT NULL,
    IN p_team_id integer DEFAULT NULL,
    IN p_assignee_id integer DEFAULT NULL,
    IN p_creator_id integer DEFAULT NULL,
    IN p_due_date date DEFAULT NULL,
    IN p_parent_task_id integer DEFAULT NULL,
    IN p_changed_by_id integer DEFAULT NULL,
    IN p_reason text DEFAULT NULL
)
    LANGUAGE plpgsql
AS $$
BEGIN
    IF NOT EXISTS (SELECT * FROM tasks."Tasks" WHERE id = p_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_id;
    END IF;
    IF p_changed_by_id IS NULL THEN
        RAISE EXCEPTION 'p_changed_by_id не может быть NULL';
    END IF;
    IF p_status_id IS NOT NULL THEN
        CALL tasks."UpdateTaskStatus"(p_id, p_status_id, p_changed_by_id, p_reason);
    END IF;
    IF p_due_date IS NOT NULL THEN
        CALL tasks."UpdateTaskDeadline"(p_id, p_due_date, p_changed_by_id, p_reason);
    END IF;
    UPDATE tasks."Tasks"
    SET
        title = COALESCE(p_title, title),
        description = COALESCE(p_description, description),
        create_date = COALESCE(p_create_date, create_date),
        team_id = COALESCE(p_team_id, team_id),
        assignee_id = COALESCE(p_assignee_id, assignee_id),
        creator_id = COALESCE(p_creator_id, creator_id),
        parent_task_id = COALESCE(p_parent_task_id, parent_task_id),
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_id;
END;
$$;
ALTER PROCEDURE tasks."UpdateTask"(integer, varchar, text, integer, date, integer, integer, integer, date, integer, integer, text) OWNER TO postgres;

-- Процедура RemoveTask
CREATE OR REPLACE PROCEDURE tasks."RemoveTask"(IN p_id integer)
    LANGUAGE plpgsql
AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM tasks."Tasks" WHERE parent_task_id = p_id) THEN
        RAISE NOTICE 'Задача % имеет подзадачи. Удаление отменено.', p_id;
        RETURN;
    END IF;
    DELETE FROM tasks."Tasks" WHERE id = p_id;
    IF NOT FOUND THEN
        RAISE NOTICE 'Задача с id % не найдена', p_id;
    END IF;
END;
$$;
ALTER PROCEDURE tasks."RemoveTask"(integer) OWNER TO postgres;

-- Процедура AssignTaskToAllMembers

    IN p_title varchar,
    IN p_description text,
    IN p_team_id integer,
    IN p_creator_id integer,
    IN p_due_date date,
    IN p_status_id integer DEFAULT 1
)
    LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO tasks."Tasks" (
        title, description, status_id, due_date,
        team_id, creator_id, assignee_id, create_date
    )

        p_title, p_description, p_status_id, p_due_date,

    IF NOT FOUND THEN
        RAISE NOTICE 'В команде % нет активных участников', p_team_id;
    END IF;
END;
$$;
ALTER PROCEDURE tasks."AssignTaskToAllMembers"(varchar, text, integer, integer, date, integer) OWNER TO postgres;

-- Процедура AddUserToTeam (в схеме teams)
CREATE OR REPLACE PROCEDURE teams."AddUserToTeam"(
    IN p_team_id integer,
    IN p_account_id integer,
    IN p_added_by_id integer
)
    LANGUAGE plpgsql
AS $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM teams."Teams" WHERE id = p_team_id) THEN
        RAISE EXCEPTION 'Команда с id % не найдена', p_team_id;

-- ---------- Схема accounts ----------
CREATE OR REPLACE FUNCTION accounts."RegisterAccount"(
    p_login varchar,
    p_password_hash varchar,
    p_email varchar,
    p_fullname text,
    p_position varchar DEFAULT NULL
) RETURNS integer
    LANGUAGE plpgsql
AS $$
DECLARE
    v_result INTEGER;
BEGIN
    IF EXISTS (SELECT 1 FROM accounts."Accounts" WHERE login = p_login) THEN
        RETURN 0;
    END IF;
    IF EXISTS (SELECT 1 FROM accounts."Accounts" WHERE email = p_email) THEN
        RETURN -1;
    END IF;
    INSERT INTO accounts."Accounts" (login, password_hash, email, fullname, "position", "role", is_active)
    VALUES (p_login, p_password_hash, p_email, p_fullname, p_position, 'User', true);
    RETURN 1;
END;
$$;
ALTER FUNCTION accounts."RegisterAccount"(varchar, varchar, varchar, text, varchar) OWNER TO postgres;

CREATE OR REPLACE FUNCTION accounts."AuthenticateAccount"(
    p_login varchar,
    p_password_hash varchar
)
    RETURNS TABLE(account_id integer, "role" varchar, status integer)
    LANGUAGE plpgsql
AS $$
DECLARE
    v_account RECORD;
BEGIN
    SELECT a.id, a."role", a.is_active INTO v_account
    FROM accounts."Accounts" a
    WHERE a.login = p_login AND a.password_hash = p_password_hash;

    IF NOT FOUND THEN
        RETURN QUERY SELECT NULL::INT, NULL::VARCHAR, 0;
        RETURN;
    END IF;
    IF v_account.is_active = false THEN
        RETURN QUERY SELECT v_account.id, v_account."role", -1;
        RETURN;
    END IF;
    RETURN QUERY SELECT v_account.id, v_account."role", 1;
END;
$$;
ALTER FUNCTION accounts."AuthenticateAccount"(varchar, varchar) OWNER TO postgres;

CREATE OR REPLACE FUNCTION accounts."GetUserStats"(p_user_id integer)
    RETURNS TABLE(total_tasks bigint, completed_tasks bigint, overdue_tasks bigint)
    LANGUAGE plpgsql
AS $$
DECLARE
    accepted_status_id INTEGER;
BEGIN
    SELECT id INTO accepted_status_id FROM statuses."Statuses" WHERE name = 'Принята' LIMIT 1;
    IF accepted_status_id IS NULL THEN
        RETURN QUERY SELECT 0::BIGINT, 0::BIGINT, 0::BIGINT;
        RETURN;
    END IF;
    RETURN QUERY
        SELECT
            COUNT(*)::BIGINT AS total_tasks,
            COUNT(*) FILTER (WHERE t.status_id = accepted_status_id)::BIGINT AS completed_tasks,
            COUNT(*) FILTER (WHERE t.due_date < CURRENT_DATE AND t.status_id != accepted_status_id)::BIGINT AS overdue_tasks
        FROM tasks."Tasks" t
        WHERE t.assignee_id = p_user_id;
END;
$$;
ALTER FUNCTION accounts."GetUserStats"(integer) OWNER TO postgres;

-- ---------- Схема tasks ----------
CREATE OR REPLACE FUNCTION tasks."GetUserTasks"(p_account_id integer)
    RETURNS TABLE(id integer, title varchar, status_name varchar, due_date date, is_personal boolean)
    LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id, t.title, s.name AS status_name, t.due_date, TRUE AS is_personal
        FROM tasks."Tasks" t
                 JOIN statuses."Statuses" s ON t.status_id = s.id
        WHERE t.assignee_id = p_account_id
        UNION
        SELECT t.id, t.title, s.name AS status_name, t.due_date, FALSE AS is_personal
        FROM tasks."Tasks" t
                 JOIN statuses."Statuses" s ON t.status_id = s.id
                 JOIN teams."TeamMembers" tm ON t.team_id = tm.team_id
        WHERE tm.account_id = p_account_id AND t.assignee_id IS NULL;
END;
$$;
ALTER FUNCTION tasks."GetUserTasks"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION tasks."GetTaskDetails"(p_task_id integer)
    RETURNS TABLE(id integer, title varchar, description text, due_date date,
                  status_name varchar, creator_fullname varchar)
    LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id::INT, t.title::VARCHAR(200), t.description::TEXT, t.due_date::DATE,
               s.name::VARCHAR(50) AS status_name, a.fullname::VARCHAR(150) AS creator_fullname
        FROM tasks."Tasks" t
                 JOIN statuses."Statuses" s ON t.status_id = s.id
                 JOIN accounts."Accounts" a ON t.creator_id = a.id
        WHERE t.id = p_task_id;
END;
$$;
ALTER FUNCTION tasks."GetTaskDetails"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION tasks."CopyTask"(
    p_task_id integer,
    p_new_creator_id integer,
    p_new_assignee_id integer,
    p_copy_subtasks boolean
) RETURNS integer
    LANGUAGE plpgsql
AS $$
DECLARE
    new_task_id INTEGER;
    subtask RECORD;
BEGIN
    INSERT INTO tasks."Tasks" (
        title, description, status_id, create_date, due_date,
        team_id, assignee_id, creator_id, parent_task_id, created_at, updated_at
    )
    SELECT
        title, description, status_id, CURRENT_DATE, due_date,
        team_id, p_new_assignee_id, p_new_creator_id, NULL, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP
    FROM tasks."Tasks"
    WHERE id = p_task_id
    RETURNING id INTO new_task_id;

    IF p_copy_subtasks THEN
        FOR subtask IN
            SELECT id FROM tasks."Tasks" WHERE parent_task_id = p_task_id
            LOOP
                PERFORM tasks."CopyTask"(subtask.id, p_new_creator_id, p_new_assignee_id, true);
                UPDATE tasks."Tasks" SET parent_task_id = new_task_id
                WHERE parent_task_id = p_task_id
                  AND id = (SELECT max(id) FROM tasks."Tasks"
                            WHERE title = (SELECT title FROM tasks."Tasks" WHERE id = subtask.id)
                            ORDER BY id DESC LIMIT 1);
            END LOOP;
    END IF;
    RETURN new_task_id;
END;
$$;
ALTER FUNCTION tasks."CopyTask"(integer, integer, integer, boolean) OWNER TO postgres;

CREATE OR REPLACE FUNCTION tasks."GetOverdueTasks"(p_account_id integer DEFAULT NULL)
    RETURNS TABLE(id integer, title varchar, description text, due_date date,
                  status_name varchar, assignee_name text, overdue_days integer)
    LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id, t.title, t.description, t.due_date,
               s.name AS status_name, a.fullname AS assignee_name,
               (CURRENT_DATE - t.due_date)::INTEGER AS overdue_days
        FROM tasks."Tasks" t
                 JOIN statuses."Statuses" s ON t.status_id = s.id
                 LEFT JOIN accounts."Accounts" a ON t.assignee_id = a.id
        WHERE t.due_date < CURRENT_DATE
          AND s.name != 'Принята'
          AND (p_account_id IS NULL OR t.assignee_id = p_account_id)
        ORDER BY t.due_date;
END;
$$;
ALTER FUNCTION tasks."GetOverdueTasks"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION tasks."SearchTasks"(
    p_query text,
    p_user_id integer,
    p_role varchar
)
    RETURNS TABLE(id integer, title varchar, description text, due_date date,
                  status_name varchar, assignee_name text, relevance real)
    LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT
            t.id, t.title, t.description, t.due_date,
            s.name AS status_name, a.fullname AS assignee_name,
            (similarity(t.title, p_query) + similarity(COALESCE(t.description, ''), p_query))::REAL AS relevance
        FROM tasks."Tasks" t
                 JOIN statuses."Statuses" s ON t.status_id = s.id
                 LEFT JOIN accounts."Accounts" a ON t.assignee_id = a.id
        WHERE
            (t.title ILIKE '%' || p_query || '%' OR t.description ILIKE '%' || p_query || '%')
          AND (
            p_role = 'Admin'
                OR (p_role = 'TeamLead' AND (t.creator_id = p_user_id OR t.assignee_id = p_user_id))
                OR (p_role = 'User' AND t.assignee_id = p_user_id)
            )
        ORDER BY relevance DESC;
END;
$$;
ALTER FUNCTION tasks."SearchTasks"(text, integer, varchar) OWNER TO postgres;
