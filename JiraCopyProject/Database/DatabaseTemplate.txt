-- 1. Accounts

CREATE OR REPLACE PROCEDURE "UpdateTaskStatus"(
    IN p_task_id integer,
    IN p_new_status integer,
    IN p_changed_by_id integer,
    IN p_reason text DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_old_status_id INT;
BEGIN
    IF NOT EXISTS (SELECT * FROM "Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;

    SELECT status_id INTO v_old_status_id FROM "Tasks" WHERE id = p_task_id;
    IF v_old_status_id = p_new_status THEN
        RAISE NOTICE 'Статус задачи % не изменился', p_task_id;
        RETURN;
    END IF;
    UPDATE "Tasks"
    SET status_id = p_new_status,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;
    INSERT INTO "TaskStatusHistory"(
        task_id, old_status_id, new_status_id, changed_by_id, reason, changed_at
    ) VALUES (
        p_task_id, v_old_status_id, p_new_status, p_changed_by_id, p_reason, CURRENT_TIMESTAMP
    );
END;
$$;
ALTER PROCEDURE "UpdateTaskStatus"(integer, integer, integer, text) OWNER TO postgres;

CREATE OR REPLACE PROCEDURE "UpdateTaskDeadline"(
    IN p_task_id integer,
    IN p_new_deadline date,
    IN p_changed_by_id integer,
    IN p_reason text DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_old_deadline DATE;
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;
    SELECT due_date INTO v_old_deadline FROM "Tasks" WHERE id = p_task_id;
    IF v_old_deadline = p_new_deadline THEN
        RAISE NOTICE 'Дедлайн задачи % не изменился', p_task_id;
        RETURN;
    END IF;
    UPDATE "Tasks"
    SET due_date = p_new_deadline,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;
    INSERT INTO "TaskDeadLineHistory"(
        task_id, old_due_date, new_due_date, changed_by_id, reason, changed_at
    ) VALUES (
        p_task_id, v_old_deadline, p_new_deadline, p_changed_by_id, p_reason, CURRENT_TIMESTAMP
    );
END;
$$;
ALTER PROCEDURE "UpdateTaskDeadline"(integer, date, integer, text) OWNER TO postgres;

CREATE OR REPLACE PROCEDURE "UpdateTask"(
    IN p_id integer,
    IN p_title varchar DEFAULT NULL,
    IN p_description text DEFAULT NULL,
    IN p_status_id integer DEFAULT NULL,
    IN p_create_date date DEFAULT NULL,
    IN p_team_id integer DEFAULT NULL,
    IN p_assignee_id integer DEFAULT NULL,
    IN p_creator_id integer DEFAULT NULL,
    IN p_due_date date DEFAULT NULL,
    IN p_parent_task_id integer DEFAULT NULL,
    IN p_changed_by_id integer DEFAULT NULL,
    IN p_reason text DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
BEGIN
    IF NOT EXISTS (SELECT * FROM "Tasks" WHERE id = p_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_id;
    END IF;
    IF p_changed_by_id IS NULL THEN
        RAISE EXCEPTION 'p_changed_by_id не может быть NULL';
    END IF;
    IF p_status_id IS NOT NULL THEN
        CALL "UpdateTaskStatus"(p_id, p_status_id, p_changed_by_id, p_reason);
    END IF;
    IF p_due_date IS NOT NULL THEN
        CALL "UpdateTaskDeadline"(p_id, p_due_date, p_changed_by_id, p_reason);
    END IF;
    UPDATE "Tasks"
    SET
        title = COALESCE(p_title, title),
        description = COALESCE(p_description, description),
        create_date = COALESCE(p_create_date, create_date),
        team_id = COALESCE(p_team_id, team_id),
        assignee_id = COALESCE(p_assignee_id, assignee_id),
        creator_id = COALESCE(p_creator_id, creator_id),
        parent_task_id = COALESCE(p_parent_task_id, parent_task_id),
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_id;
END;
$$;
ALTER PROCEDURE "UpdateTask"(integer, varchar, text, integer, date, integer, integer, integer, date, integer, integer, text) OWNER TO postgres;

CREATE OR REPLACE PROCEDURE "RemoveTask"(IN p_id integer)
LANGUAGE plpgsql
AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM "Tasks" WHERE parent_task_id = p_id) THEN
        RAISE NOTICE 'Задача % имеет подзадачи. Удаление отменено.', p_id;
        RETURN;
    END IF;
    DELETE FROM "Tasks" WHERE id = p_id;
    IF NOT FOUND THEN
        RAISE NOTICE 'Задача с id % не найдена', p_id;
    END IF;
END;
$$;
ALTER PROCEDURE "RemoveTask"(integer) OWNER TO postgres;

    IN p_team_id integer,

    IN p_added_by_id integer
)

AS $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Teams" WHERE id = p_team_id) THEN
        RAISE EXCEPTION 'Команда с id % не найдена', p_team_id;

CREATE OR REPLACE FUNCTION "RegisterAccount"(
    p_login varchar,
    p_password_hash varchar,
    p_email varchar,
    p_fullname text,
    p_position varchar DEFAULT NULL
) RETURNS integer
LANGUAGE plpgsql
AS $$
DECLARE
    v_result INTEGER;
BEGIN
    IF EXISTS (SELECT 1 FROM "Accounts" WHERE login = p_login) THEN
        RETURN 0;
    END IF;
    IF EXISTS (SELECT 1 FROM "Accounts" WHERE email = p_email) THEN
        RETURN -1;
    END IF;
    INSERT INTO "Accounts" (login, password_hash, email, fullname, position, role, is_active)
    VALUES (p_login, p_password_hash, p_email, p_fullname, p_position, 'User', true);
    RETURN 1;
END;
$$;
ALTER FUNCTION "RegisterAccount"(varchar, varchar, varchar, text, varchar) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "GetUserTasks"(p_account_id integer)
RETURNS TABLE(id integer, title varchar, status_name varchar, due_date date, is_personal boolean)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id, t.title, s.name AS status_name, t.due_date, TRUE AS is_personal
        FROM "Tasks" t
            JOIN "Statuses" s ON t.status_id = s.id
        WHERE t.assignee_id = p_account_id
        UNION
        SELECT t.id, t.title, s.name AS status_name, t.due_date, FALSE AS is_personal
        FROM "Tasks" t
            JOIN "Statuses" s ON t.status_id = s.id
            JOIN "TeamMembers" tm ON t.team_id = tm.team_id
        WHERE tm.account_id = p_account_id AND t.assignee_id IS NULL;
END;
$$;
ALTER FUNCTION "GetUserTasks"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "AuthenticateAccount"(
    p_login varchar,
    p_password_hash varchar
)
RETURNS TABLE(account_id integer, role varchar, status integer)
LANGUAGE plpgsql
AS $$
DECLARE
    v_account RECORD;
BEGIN
    SELECT a.id, a.role, a.is_active INTO v_account
    FROM "Accounts" a
    WHERE a.login = p_login AND a.password_hash = p_password_hash;

    IF NOT FOUND THEN
        RETURN QUERY SELECT NULL::INT, NULL::VARCHAR, 0;
        RETURN;
    END IF;
    IF v_account.is_active = false THEN
        RETURN QUERY SELECT v_account.id, v_account.role, -1;
        RETURN;
    END IF;
    RETURN QUERY SELECT v_account.id, v_account.role, 1;
END;
$$;
ALTER FUNCTION "AuthenticateAccount"(varchar, varchar) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "GetUserStats"(p_user_id integer)
RETURNS TABLE(total_tasks bigint, completed_tasks bigint, overdue_tasks bigint)
LANGUAGE plpgsql
AS $$
DECLARE
    accepted_status_id INTEGER;
BEGIN
    SELECT id INTO accepted_status_id FROM "Statuses" WHERE name = 'Принята' LIMIT 1;
    IF accepted_status_id IS NULL THEN
        RETURN QUERY SELECT 0::BIGINT, 0::BIGINT, 0::BIGINT;
        RETURN;
    END IF;
    RETURN QUERY
        SELECT
            COUNT(*)::BIGINT AS total_tasks,
            COUNT(*) FILTER (WHERE t.status_id = accepted_status_id)::BIGINT AS completed_tasks,
            COUNT(*) FILTER (WHERE t.due_date < CURRENT_DATE AND t.status_id != accepted_status_id)::BIGINT AS overdue_tasks
        FROM "Tasks" t
        WHERE t.assignee_id = p_user_id;
END;
$$;
ALTER FUNCTION "GetUserStats"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "CopyTask"(
    p_task_id integer,
    p_new_creator_id integer,
    p_new_assignee_id integer,
    p_copy_subtasks boolean
) RETURNS integer
LANGUAGE plpgsql
AS $$
DECLARE
    new_task_id INTEGER;
    subtask RECORD;
BEGIN
    INSERT INTO "Tasks" (
        title, description, status_id, create_date, due_date,
        team_id, assignee_id, creator_id, parent_task_id, created_at, updated_at
    )
    SELECT
        title, description, status_id, CURRENT_DATE, due_date,
        team_id, p_new_assignee_id, p_new_creator_id, NULL, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP
    FROM "Tasks"
    WHERE id = p_task_id
    RETURNING id INTO new_task_id;

    IF p_copy_subtasks THEN
        FOR subtask IN
            SELECT id FROM "Tasks" WHERE parent_task_id = p_task_id
        LOOP
            PERFORM "CopyTask"(subtask.id, p_new_creator_id, p_new_assignee_id, true);
            UPDATE "Tasks" SET parent_task_id = new_task_id
            WHERE parent_task_id = p_task_id
              AND id = (SELECT max(id) FROM "Tasks"
                        WHERE title = (SELECT title FROM "Tasks" WHERE id = subtask.id)
                        ORDER BY id DESC LIMIT 1);
        END LOOP;
    END IF;
    RETURN new_task_id;
END;
$$;
ALTER FUNCTION "CopyTask"(integer, integer, integer, boolean) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "GetTasksByTag"(p_tag_id integer)
RETURNS TABLE(id integer, title varchar, description text, due_date date,
              status_name varchar, assignee_name text)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id, t.title, t.description, t.due_date,
               s.name AS status_name, a.fullname AS assignee_name
        FROM "Tasks" t
            JOIN "TaskTags" tt ON t.id = tt.task_id
            JOIN "Statuses" s ON t.status_id = s.id
            LEFT JOIN "Accounts" a ON t.assignee_id = a.id
        WHERE tt.tag_id = p_tag_id
        ORDER BY t.due_date;
END;
$$;
ALTER FUNCTION "GetTasksByTag"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "GetOverdueTasks"(p_account_id integer DEFAULT NULL)
RETURNS TABLE(id integer, title varchar, description text, due_date date,
              status_name varchar, assignee_name text, overdue_days integer)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT t.id, t.title, t.description, t.due_date,
               s.name AS status_name, a.fullname AS assignee_name,
               (CURRENT_DATE - t.due_date)::INTEGER AS overdue_days
        FROM "Tasks" t
            JOIN "Statuses" s ON t.status_id = s.id
            LEFT JOIN "Accounts" a ON t.assignee_id = a.id
        WHERE t.due_date < CURRENT_DATE
          AND s.name != 'Принята'
          AND (p_account_id IS NULL OR t.assignee_id = p_account_id)
        ORDER BY t.due_date;
END;
$$;
ALTER FUNCTION "GetOverdueTasks"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "GetTaskStatusHistory"(p_task_id integer)
RETURNS TABLE(old_status varchar, new_status varchar, changed_by text, reason text, changed_at date)
LANGUAGE sql
AS $$
    SELECT os.name AS old_status, ns.name AS new_status, a.fullname AS changed_by, h.reason, h.changed_at
    FROM "TaskStatusHistory" h
        JOIN "Statuses" os ON h.old_status_id = os.id
        JOIN "Statuses" ns ON h.new_status_id = ns.id
        JOIN "Accounts" a ON h.changed_by_id = a.id
    WHERE h.task_id = p_task_id
    ORDER BY h.changed_at DESC;
$$;
ALTER FUNCTION "GetTaskStatusHistory"(integer) OWNER TO postgres;

CREATE OR REPLACE FUNCTION "SearchTasks"(
    p_query text,
    p_user_id integer,
    p_role varchar
)
RETURNS TABLE(id integer, title varchar, description text, due_date date,
              status_name varchar, assignee_name text, relevance real)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
        SELECT
            t.id, t.title, t.description, t.due_date,
            s.name AS status_name, a.fullname AS assignee_name,
            (similarity(t.title, p_query) + similarity(COALESCE(t.description, ''), p_query))::REAL AS relevance
        FROM "Tasks" t
            JOIN "Statuses" s ON t.status_id = s.id
            LEFT JOIN "Accounts" a ON t.assignee_id = a.id
        WHERE
            (t.title ILIKE '%' || p_query || '%' OR t.description ILIKE '%' || p_query || '%')
          AND (
            p_role = 'Admin'
               OR (p_role = 'TeamLead' AND (t.creator_id = p_user_id OR t.assignee_id = p_user_id))
               OR (p_role = 'User' AND t.assignee_id = p_user_id)
          )
        ORDER BY relevance DESC;
END;
$$;
ALTER FUNCTION "SearchTasks"(text, integer, varchar) OWNER TO postgres;
