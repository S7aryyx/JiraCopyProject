-- 2. Таблицы и индексы

CREATE TABLE "Accounts" (
    id            SERIAL PRIMARY KEY,
    login         VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    email         VARCHAR(150) NOT NULL UNIQUE,
    fullname      TEXT NOT NULL,
    position      VARCHAR(100),
    role          VARCHAR(20),
    created_at    DATE DEFAULT CURRENT_TIMESTAMP,
    is_active     BOOLEAN DEFAULT TRUE
);

CREATE INDEX idx_accounts_login ON "Accounts" (login);
CREATE INDEX idx_accounts_email ON "Accounts" (email);
CREATE INDEX idx_accounts_role ON "Accounts" (role);

CREATE TABLE "Teams" (
    id           SERIAL PRIMARY KEY,
    name         VARCHAR(100) NOT NULL,
    description  TEXT,
    team_lead_id INTEGER REFERENCES "Accounts" ON DELETE SET NULL,
    created_at   DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_teams_team_lead_id ON "Teams" (team_lead_id);

CREATE TABLE "TeamMembers" (
    team_id    INTEGER NOT NULL REFERENCES "Teams" ON DELETE CASCADE,
    account_id INTEGER NOT NULL REFERENCES "Accounts" ON DELETE CASCADE,
    created_at DATE DEFAULT CURRENT_TIMESTAMP,
    is_active  BOOLEAN DEFAULT TRUE,
    PRIMARY KEY (team_id, account_id)
);

CREATE TABLE "Statuses" (
    id          SERIAL PRIMARY KEY,
    name        VARCHAR(50) NOT NULL UNIQUE,
    code        VARCHAR(50) NOT NULL UNIQUE,
    description TEXT,
    sort_order  INTEGER DEFAULT 0
);

CREATE INDEX idx_statuses_code ON "Statuses" (code);

CREATE TABLE "Tasks" (
    id             SERIAL PRIMARY KEY,
    title          VARCHAR(200) NOT NULL,
    description    TEXT,
    status_id      INTEGER NOT NULL REFERENCES "Statuses",
    create_date    DATE DEFAULT CURRENT_TIMESTAMP,
    due_date       DATE NOT NULL,
    team_id        INTEGER REFERENCES "Teams" ON DELETE SET NULL,
    assignee_id    INTEGER REFERENCES "Accounts" ON DELETE SET NULL,
    creator_id     INTEGER NOT NULL REFERENCES "Accounts",
    parent_task_id INTEGER REFERENCES "Tasks" ON DELETE CASCADE,
    created_at     DATE DEFAULT CURRENT_TIMESTAMP,
    updated_at     DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_tasks_status_id ON "Tasks" (status_id);
CREATE INDEX idx_tasks_assignee_id ON "Tasks" (assignee_id);
CREATE INDEX idx_tasks_creator_id ON "Tasks" (creator_id);
CREATE INDEX idx_tasks_parent_task_id ON "Tasks" (parent_task_id);
CREATE INDEX idx_tasks_due_date ON "Tasks" (due_date);
CREATE INDEX idx_tasks_assignee_status ON "Tasks" (assignee_id, status_id);
CREATE INDEX idx_tasks_creator_status ON "Tasks" (creator_id, status_id);
CREATE INDEX idx_tasks_team_status ON "Tasks" (team_id, status_id);

CREATE TABLE "TaskDeadLineHistory" (
    id            SERIAL PRIMARY KEY,
    task_id       INTEGER NOT NULL REFERENCES "Tasks" ON DELETE CASCADE,
    old_due_date  DATE NOT NULL,
    new_due_date  DATE NOT NULL,
    changed_by_id INTEGER NOT NULL REFERENCES "Accounts",
    reason        TEXT,
    changed_at    DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_taskdeadlinehistory_task_id ON "TaskDeadLineHistory" (task_id);
CREATE INDEX idx_taskdeadlinehistory_changed_by_id ON "TaskDeadLineHistory" (changed_by_id);
CREATE INDEX idx_task_deadline_task_date ON "TaskDeadLineHistory" (task_id ASC, changed_at DESC);

CREATE TABLE "TaskStatusHistory" (
    id            SERIAL PRIMARY KEY,
    task_id       INTEGER NOT NULL REFERENCES "Tasks" ON DELETE CASCADE,
    old_status_id INTEGER NOT NULL REFERENCES "Statuses",
    new_status_id INTEGER NOT NULL REFERENCES "Statuses",
    changed_by_id INTEGER NOT NULL REFERENCES "Accounts",
    reason        TEXT,
    changed_at    DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_taskstatushistory_task_id ON "TaskStatusHistory" (task_id);
CREATE INDEX idx_taskstatushistory_changed_by_id ON "TaskStatusHistory" (changed_by_id);
CREATE INDEX idx_task_history_task_date ON "TaskStatusHistory" (task_id ASC, changed_at DESC);

-- Исправлено название поля autor_id → author_id
CREATE TABLE "TaskComments" (
    id         SERIAL PRIMARY KEY,
    task_id    INTEGER NOT NULL REFERENCES "Tasks" ON DELETE CASCADE,
    author_id  INTEGER NOT NULL REFERENCES "Accounts",
    comment    TEXT NOT NULL,
    created_at DATE DEFAULT CURRENT_TIMESTAMP,
    updated_at DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_taskcomments_task_id ON "TaskComments" (task_id);
CREATE INDEX idx_taskcomments_author_id ON "TaskComments" (author_id);

CREATE TABLE "Tags" (
    id         SERIAL PRIMARY KEY,
    name       VARCHAR(50) NOT NULL UNIQUE,
    color      VARCHAR(7) DEFAULT '#FFFFFF',
    created_at DATE DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE "TaskTags" (
    task_id INTEGER NOT NULL REFERENCES "Tasks" ON DELETE CASCADE,
    tag_id  INTEGER NOT NULL REFERENCES "Tags" ON DELETE SET NULL,
    PRIMARY KEY (task_id, tag_id)
);

-- 1 Создание задачи (исправлено: добавлен параметр p_team_id)
CREATE OR REPLACE PROCEDURE "InsertTask"(
    IN p_title VARCHAR,
    IN p_description TEXT,
    IN p_status_id INTEGER,
    IN p_assignee_id INTEGER,
    IN p_creator_id INTEGER,
    IN p_team_id INTEGER DEFAULT NULL,
    IN p_due_date DATE DEFAULT NULL,
    IN p_parent_task_id INTEGER DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO "Tasks" (
        title,
        description,
        status_id,
        due_date,
        assignee_id,
        creator_id,
        team_id,
        parent_task_id,
        create_date
    )
    VALUES (
        p_title,
        p_description,
        p_status_id,
        COALESCE(p_due_date, CURRENT_DATE),
        p_assignee_id,
        p_creator_id,
        p_team_id,
        p_parent_task_id,
        CURRENT_DATE
    );
END;
$$;

-- 2 Обновление статуса задачи
CREATE OR REPLACE PROCEDURE "UpdateTaskStatus"(
    IN p_task_id INTEGER,
    IN p_new_status INTEGER,
    IN p_changed_by_id INTEGER,
    IN p_reason TEXT DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_old_status_id INT;
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;

    SELECT status_id INTO v_old_status_id FROM "Tasks" WHERE id = p_task_id;

    IF v_old_status_id = p_new_status THEN
        RAISE NOTICE 'Статус задачи % не изменился', p_task_id;
        RETURN;
    END IF;

    UPDATE "Tasks"
    SET status_id = p_new_status,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;

    INSERT INTO "TaskStatusHistory"(
        task_id,
        old_status_id,
        new_status_id,
        changed_by_id,
        reason,
        changed_at
    ) VALUES (
        p_task_id,
        v_old_status_id,
        p_new_status,
        p_changed_by_id,
        p_reason,
        CURRENT_TIMESTAMP
    );
END;
$$;

-- 3 Авто-Логгер истории даты сдачи
CREATE OR REPLACE PROCEDURE "UpdateTaskDeadline"(
    IN p_task_id INTEGER,
    IN p_new_deadline DATE,
    IN p_changed_by_id INTEGER,
    IN p_reason TEXT DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_old_deadline DATE;
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Tasks" WHERE id = p_task_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_task_id;
    END IF;

    SELECT due_date INTO v_old_deadline FROM "Tasks" WHERE id = p_task_id;

    IF v_old_deadline = p_new_deadline THEN
        RAISE NOTICE 'Дедлайн задачи % не изменился', p_task_id;
        RETURN;
    END IF;

    UPDATE "Tasks"
    SET due_date = p_new_deadline,
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_task_id;

    INSERT INTO "TaskDeadLineHistory"(
        task_id,
        old_due_date,
        new_due_date,
        changed_by_id,
        reason,
        changed_at
    ) VALUES (
        p_task_id,
        v_old_deadline,
        p_new_deadline,
        p_changed_by_id,
        p_reason,
        CURRENT_TIMESTAMP
    );
END;
$$;

-- 4 Обновление полей задачи
CREATE OR REPLACE PROCEDURE "UpdateTask"(
    IN p_id INTEGER,
    IN p_title VARCHAR DEFAULT NULL,
    IN p_description TEXT DEFAULT NULL,
    IN p_status_id INTEGER DEFAULT NULL,
    IN p_create_date DATE DEFAULT NULL,
    IN p_team_id INTEGER DEFAULT NULL,
    IN p_assignee_id INTEGER DEFAULT NULL,
    IN p_creator_id INTEGER DEFAULT NULL,
    IN p_due_date DATE DEFAULT NULL,
    IN p_parent_task_id INTEGER DEFAULT NULL,
    IN p_changed_by_id INTEGER DEFAULT NULL,
    IN p_reason TEXT DEFAULT NULL
)
LANGUAGE plpgsql
AS $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Tasks" WHERE id = p_id) THEN
        RAISE EXCEPTION 'Задача с id % не найдена', p_id;
    END IF;

    IF p_changed_by_id IS NULL THEN
        RAISE EXCEPTION 'p_changed_by_id не может быть NULL';
    END IF;

    IF p_status_id IS NOT NULL THEN
        CALL "UpdateTaskStatus"(p_id, p_status_id, p_changed_by_id, p_reason);
    END IF;

    IF p_due_date IS NOT NULL THEN
        CALL "UpdateTaskDeadline"(p_id, p_due_date, p_changed_by_id, p_reason);
    END IF;

    UPDATE "Tasks"
    SET
        title = COALESCE(p_title, title),
        description = COALESCE(p_description, description),
        create_date = COALESCE(p_create_date, create_date),
        team_id = COALESCE(p_team_id, team_id),
        assignee_id = COALESCE(p_assignee_id, assignee_id),
        creator_id = COALESCE(p_creator_id, creator_id),
        parent_task_id = COALESCE(p_parent_task_id, parent_task_id),
        updated_at = CURRENT_TIMESTAMP
    WHERE id = p_id;
END;
$$;

-- 5 Удаление задачи
CREATE OR REPLACE PROCEDURE "RemoveTask"(IN p_id INTEGER)
LANGUAGE plpgsql
AS $$
BEGIN
    IF EXISTS (SELECT 1 FROM "Tasks" WHERE parent_task_id = p_id) THEN
        RAISE NOTICE 'Задача % имеет подзадачи. Удаление отменено.', p_id;
        RETURN;
    END IF;

    DELETE FROM "Tasks" WHERE id = p_id;

    IF NOT FOUND THEN
        RAISE NOTICE 'Задача с id % не найдена', p_id;
    END IF;
END;
$$;

-- Добавление пользователя в команду
CREATE OR REPLACE PROCEDURE "AddUserToTeam"(
    IN p_team_id INTEGER,
    IN p_account_id INTEGER,
    IN p_added_by_id INTEGER
)
LANGUAGE plpgsql
AS $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM "Teams" WHERE id = p_team_id) THEN
        RAISE EXCEPTION 'Команда с id % не найдена', p_team_id;
    END IF;

    IF NOT EXISTS (SELECT 1 FROM "Accounts" WHERE id = p_account_id) THEN
        RAISE EXCEPTION 'Пользователь с id % не найден', p_account_id;
    END IF;

    INSERT INTO "TeamMembers" (team_id, account_id, created_at, is_active)
    VALUES (p_team_id, p_account_id, CURRENT_DATE, true)
    ON CONFLICT (team_id, account_id) DO NOTHING;

    RAISE NOTICE 'Пользователь % добавлен в команду %', p_account_id, p_team_id;
END;
$$;

-- Чистка всех таблиц
CREATE OR REPLACE PROCEDURE "ClearAll"()
LANGUAGE plpgsql
AS $$
BEGIN
    TRUNCATE TABLE "TaskDeadLineHistory", "TaskStatusHistory", "TaskComments",
        "TaskTags", "TeamMembers", "Tasks", "Tags", "Teams",
        "Accounts", "Statuses"
        RESTART IDENTITY CASCADE;

    RAISE NOTICE 'Все данные очищены';
END;
$$;

-- 8 Регистрация нового аккаунта
CREATE OR REPLACE FUNCTION "RegisterAccount"
(
    p_login VARCHAR(100),
    p_password_hash VARCHAR(255),
    p_email VARCHAR(150),
    p_fullname TEXT,
    p_position VARCHAR(100) DEFAULT NULL
)
RETURNS INTEGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_result INTEGER;
BEGIN
    IF EXISTS (SELECT 1 FROM "Accounts" WHERE login = p_login) THEN
        RETURN 0;  -- Логин уже занят
    END IF;

    IF EXISTS (SELECT 1 FROM "Accounts" WHERE email = p_email) THEN
        RETURN -1; -- Email уже зарегистрирован
    END IF;
    
    INSERT INTO "Accounts" (login, password_hash, email, fullname, position, role, is_active)
    VALUES (p_login, p_password_hash, p_email, p_fullname, p_position, 'User', true);
    
    RETURN 1; -- Успех
END;
$$;

-- 9 Авторизация пользователя
CREATE OR REPLACE FUNCTION "AuthenticateAccount"
(
    p_login VARCHAR(100),
    p_password_hash VARCHAR(255)
)
RETURNS TABLE(
    account_id INT,
    role VARCHAR(20),
    status INT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_account RECORD;
BEGIN
    SELECT id, role, is_active INTO v_account
    FROM "Accounts"
    WHERE login = p_login AND password_hash = p_password_hash;
    
    IF NOT FOUND THEN
        RETURN QUERY SELECT NULL::INT, NULL::VARCHAR, 0;
        RETURN;
    END IF;
    
    IF v_account.is_active = false THEN
        RETURN QUERY SELECT v_account.id, v_account.role, -1;
        RETURN;
    END IF;
    
    RETURN QUERY SELECT v_account.id, v_account.role, 1;
END;
$$;

-- 10 Назначение задачи всем активным участникам команды (опционально)
CREATE OR REPLACE PROCEDURE "AssignTaskToAllMembers"
(
    p_title VARCHAR(200),
    p_description TEXT,
    p_team_id INT,
    p_creator_id INT,
    p_due_date DATE,
    p_status_id INT DEFAULT 1
)
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO "Tasks" (
        title,
        description,
        status_id,
        due_date,
        team_id,
        creator_id,
        assignee_id,
        create_date
    )
    SELECT
        p_title,
        p_description,
        p_status_id,
        p_due_date,
        p_team_id,
        p_creator_id,
        tm.account_id,
        CURRENT_DATE
    FROM "TeamMembers" tm
    WHERE tm.team_id = p_team_id
      AND tm.is_active = true;
    
    IF NOT FOUND THEN
        RAISE NOTICE 'В команде % нет активных участников', p_team_id;
    END IF;
END;
$$;

-- 11 Функция для получения деталей о задаче
CREATE OR REPLACE FUNCTION "GetTaskDetails"(p_task_id INT)
RETURNS TABLE(
    id INT,
    title VARCHAR(200),
    description TEXT,
    due_date DATE,
    status_name VARCHAR(50),
    creator_fullname VARCHAR(150)
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        t.id::INT,
        t.title::VARCHAR(200),
        t.description::TEXT,
        t.due_date::DATE,
        s.name::VARCHAR(50) AS status_name,
        a.fullname::VARCHAR(150) AS creator_fullname
    FROM "Tasks" t
    JOIN "Statuses" s ON t.status_id = s.id
    JOIN "Accounts" a ON t.creator_id = a.id
    WHERE t.id = p_task_id;
END;
$$;

-- 12 Функция для получения подзадач
CREATE OR REPLACE FUNCTION "GetTaskSubtasks"(p_parent_task_id INT)
RETURNS TABLE(
    id INT,
    title VARCHAR(200),
    status_name VARCHAR(50),
    due_date DATE
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    SELECT 
        t.id,
        t.title,
        s.name AS status_name,
        t.due_date
    FROM "Tasks" t
    JOIN "Statuses" s ON t.status_id = s.id
    WHERE t.parent_task_id = p_parent_task_id
    ORDER BY t.id;
END;
$$;

-- 13 Получение всех задач пользователя
CREATE OR REPLACE FUNCTION "GetUserTasks"(p_account_id INTEGER)
RETURNS TABLE(
    id INTEGER,
    title VARCHAR(200),
    status_name VARCHAR(50),
    due_date DATE,
    is_personal BOOLEAN
)
LANGUAGE plpgsql
AS $$
BEGIN
    RETURN QUERY
    -- Личные задачи            
    SELECT 
        t.id,
        t.title,
        s.name AS status_name,
        t.due_date,
        TRUE AS is_personal
    FROM "Tasks" t
    JOIN "Statuses" s ON t.status_id = s.id
    WHERE t.assignee_id = p_account_id

    UNION

    -- Командные задачи 
    SELECT 
        t.id,
        t.title,
        s.name AS status_name,
        t.due_date,
        FALSE AS is_personal
    FROM "Tasks" t
    JOIN "Statuses" s ON t.status_id = s.id
    JOIN "TeamMembers" tm ON t.team_id = tm.team_id
    WHERE tm.account_id = p_account_id
      AND t.assignee_id IS NULL;
END;
$$;
